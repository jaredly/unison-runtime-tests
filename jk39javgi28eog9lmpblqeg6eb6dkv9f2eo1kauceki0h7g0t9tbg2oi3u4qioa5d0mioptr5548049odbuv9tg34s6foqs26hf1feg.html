  base.Map.internal.Tip             -> base.Map.internal.Tip
  base.Map.internal.Bin sx kx x l r ->
    use base.Map adjustWithKey

base.Map.alter f k = cases
  base.Map.internal.Tip             ->
    use base.Optional None
    match f None with
      None                 -> base.Map.internal.Tip
      base.Optional.Some x -> base.Map.singleton k x
  base.Map.internal.Bin sx kx x l r ->
    use base.Map alter
    use base.Map.internal balance
    match base.Universal.ordering k kx with
      base.Ordering.Less    -> balance kx x (alter f k l) r
      base.Ordering.Greater -> balance kx x l (alter f k r)
      base.Ordering.Equal   ->
        use base.Optional Some
        match f (Some x) with
          Some x'            -> base.Map.internal.Bin sx kx x' l r
          base.Optional.None -> base.Map.internal.glue l r

base.Map.difference = cases
  base.Map.internal.Tip _                -> base.Map.internal.Tip
  t1 base.Map.internal.Tip               -> t1
  t1 (base.Map.internal.Bin _ k _ l2 r2) ->
    use base.Map difference size
    use base.Nat +
    use base.Universal ==
    (l1, r1) = base.Map.split k t1
    l1l2 = difference l1 l2
    r1r2 = difference r1 r2
    if size l1l2 + size r1r2 == size t1 then t1
    else base.Map.internal.link2 l1l2 r1r2

doc

A base.Map k v is a map from keys of type k to values of type v. Duplicate keys are not possible; any given key is either not present in the base.Map or it maps to exactly one value. Different keys can map to the same value.

Note that universal equality is not defined on base.Map, but two maps can be compared for equality using base.Map.==.

Constructing maps

base.Map k v

The empty base.Map.

k -> v -> base.Map k v

Constructs a new base.Map with a single key and value.

[(k, v)] -> base.Map k v

Constructs a base.Map from a base.List of key/value pairs. If the base.List contains more than one value for the same key, the last value for the key is used. See base.Map.fromListWith and base.Map.fromListWithKey if you want to combine values under duplicate keys in some other way.

(v ->{e} v ->{e} v) -> [(k, v)] ->{e} base.Map k v

Constructs a base.Map from a base.List of key/value pairs with a combining function. If the base.List contains more than one value for the same key, the values are combined using the function.

See also base.Map.fromList and base.Map.fromListWithKey

(k ->{e} v ->{e} v ->{e} v) -> [(k, v)] ->{e} base.Map k v

Constructs a base.Map from a base.List of key/value pairs with a combining function. If the base.List contains more than one value for the same key, the values are combined using the function, which can take the key into account.

See also base.Map.fromList and base.Map.fromListWith

Putting keys and values into a map

k -> v -> base.Map k v -> base.Map k v

Puts a new key and value in the base.Map.

The expression base.Map.insert k v m puts the key k and associated value v in the base.Map m. If the key k is already present in the base.Map m, the old value is replaced with v.

(v ->{e} v ->{e} v) -> k -> v -> base.Map k v ->{e} base.Map k v

The expression base.Map.putWith f k v m puts a new key k and associated value v in the base.Map m. If the key k is already present in the base.Map, the new value v is combined with the old value using the function f. The function f will receive the new value v as its first argument and the old value under k as the second argument.

For example, if

m = fromList [(5, "a"), (3, "b")]

then

base.Map.putWith (++) 5 "x" m = base.Map.fromList [(3, "b"), (5, "xa")]
base.Map.putWith (++) 7 "x" m = base.Map.fromList [(3, "b"), (5, "a"), (7, "x")]
base.Map.putWith (++) 5 "x" base.Map.empty = base.Map.singleton 5 "x"

(k ->{e} v ->{e} v ->{e} v) -> k -> v -> base.Map k v ->{e} base.Map k v

Adds a key and value to the base.Map combining the key, new value and old value with a function.

The expression base.Map.putWithKey f k v m puts a new key k and associated value v in the base.Map m. If the key k is already present in the base.Map, the new value v is combined with the key and old value using the function f. The function f will receive the new value v as its second argument and the old value under k as the third argument.

For example, if

f key new old = toText key ++ ":" ++ new ++ "|" ++ old
m = fromList [(5, "a"), (3, "b")]

then

base.Map.putWith f 5 "x" m = base.Map.fromList [(3, "b"), (5, "5:x|a")]
base.Map.putWith f 7 "x" m = base.Map.fromList [(3, "b"), (5, "a"), (7, "x")]
base.Map.putWith f 5 "x" base.Map.empty = base.Map.singleton 5 "x"

(k ->{e} v ->{e} v ->{e} v) -> k -> v -> base.Map k v ->{e} (base.Optional v, base.Map k v)

Puts a new key and value in the base.Map while also getting the old value under that key, if any.

The expression base.Map.putGetWithKey f k v m is equivalent to (base.Map.get k m, base.Map.putWithKey f k v m). It returns a pair where the first element is the value under the key k in the base.Map m (or base.Optional if k is not present) and the second element is the new base.Map after putting the value v under the key k using the function f to combine v with the old value.

Map lookup and queries

k -> base.Map k v -> base.Boolean

The expression base.Map.contains k m returns true if the key k exists in the base.Map m, and false otherwise.

k -> base.Map k v -> base.Optional v

The expression base.Map.get k m returns the value under the key k in the base.Map m, or base.Optional if the key k is not present in the base.Map.

v -> k -> base.Map k v -> v

The expression base.Map.getOrElse v k m returns the value under the key k in the base.Map m if the key k is present, or the default value v if k is not present in the base.Map m.

For example, if

m = base.Map.fromList [(1,"one"),(3,"three")]

then

getOrElse "oops" 2 m = "oops"
getOrElse "oops" 3 m = "three"

base.Map k v -> base.Optional (k, v)

Finds the maximal key of the base.Map and the value under that key. Returns base.Optional if the base.Map is empty.

base.Map k v -> base.Optional (k, v)

Finds the minimal key of the base.Map and the value under that key. Returns base.Optional if the base.Map is empty.

base.Map k v -> base.Boolean

Returns true if the base.Map is base.Map.empty, otherwise false.

base.Map k v -> base.Nat

Gets the number of elements in the base.Map.

Removing elements from the map

k -> base.Map k v -> base.Map k v

Deletes a key and its value from the base.Map. Does nothing if the key is not in the base.Map.

Modifying the data in a map

(v ->{e} v) -> k -> base.Map k v ->{e} base.Map k v

base.Map.adjust f k m updates the value under the key k in the base.Map m with the result of applying the function f to the existing value under k. Does nothing if the key k is not in the base.Map.

For example, if

a = base.Map.fromList [(5, [1,2]), (3, [4,5])]

then,

base.Map.adjust base.List.reverse 5 a = base.Map.fromList [(5, [2,1]), (3, [4,5])]
base.Map.adjust base.List.reverse 7 a = a
base.Map.adjust base.List.reverse 7 base.Map.empty = base.Map.empty

(k ->{e} v ->{e} v) -> k -> base.Map k v ->{e} base.Map k v

base.Map.adjustWithKey f k m updates the value under the key k in the base.Map m with the result of applying the function f to both k and the existing value under k. Does nothing if the key k is not in the base.Map.

For example, if

a = base.Map.fromList [(5, [1,2]), (3, [4,5])]

then,

base.Map.adjustWithKey (+:) 5 a = base.Map.fromList [(5, [5,1,2]), (3, [4,5])]
base.Map.adjustWithKey (+:) 7 a = a
base.Map.adjustWithKey (+:) 7 base.Map.empty = base.Map.empty

(base.Optional v ->{e} base.Optional v) -> k -> base.Map k v ->{e} base.Map k v

The expression base.Map.alter f k m changes the value under the key k in the base.Map m, or the absence thereof. base.Map.alter can be used to insert, delete, or update a value in a base.Map.

The functions base.Map.alter and base.Map.get stand in a naturality relationship, as follows: base.Map.get k (base.Map.alter f k m) = f (base.Map.get k m)

For example, if

f _ = None
g _ = Some "c"
m = base.Map.get [(5, "a"), (3, "b")]

then

base.Map.alter f 7 m = m
base.Map.alter f 5 m = base.Map.singleton 3 "b"
base.Map.alter g 7 m = base.Map.fromList [(5, "a"), (3, "b"), (7, "c")]
base.Map.alter g 5 m = base.Map.fromList [(3, "b"), (5, "c")]

(v ->{e} base.Optional v) -> k -> base.Map k v ->{e} base.Map k v

Updates or deletes a value under a key in a base.Map. The expression base.Map.update f k m updates the value under k in the base.Map m, if present, by passing the existing value to the function f. If f returns base.Optional, the key k is deleted from the base.Map. If f returns base.Optional x, the value under k becomes x.

For example, if

m = base.Map.fromList [(5, 10), (3, 21)]
f k x = if base.Nat.mod x 2 == 0 then base.Optional (x + 1) else base.Optional

then

base.Map.update f 5 m = base.Map.fromList [(5, 11), (3, 21)]
base.Map.update f 7 m = m
base.Map.update f 3 m = base.Map.singleton 5 10

(k ->{e} v ->{e} base.Optional v) -> k -> base.Map k v ->{e} base.Map k v

Updates or deletes a value under a key in a base.Map. The expression base.Map.updateWithKey f k m updates the value under k in the base.Map m, if present, by passing both the key k and its value to the function f. If f returns base.Optional, the key k is deleted from the base.Map. If f returns base.Optional x, the value under k becomes x.

For example, if

m = base.Map.fromList [(5, 10), (3, 21)]
f k x = if base.Nat.mod x 2 == 0 then base.Optional (x + k) else base.Optional

then

base.Map.updateWithKey f 5 m = base.Map.fromList [(5, 15), (3, 21)]
base.Map.updateWithKey f 7 m = m
base.Map.updateWithKey f 3 m = base.Map.singleton 5 10

Combining maps

base.Map k a -> base.Map k b -> base.Map k a

The expression base.Map.difference m1 m2 constructs a new base.Map with all elements (k,v) from m1 where k doesn't exist as a key in m2.

base.Map k a -> base.Map k b -> base.Map k a

The intersection of two maps. Returns data in the first map for keys existing in both maps.

For example, if

x = base.Map.fromList [(5, "a"), (3, "b")]
y = base.Map.fromList [(5, "A"), (7, "C")]

then

base.Map.intersect x y = base.Map.singleton 5 "a"

(a ->{e} b ->{e} c) -> base.Map k a -> base.Map k b ->{e} base.Map k c

Intersection with a combining function. The expression base.Map.intersectWith f m1 m2 combines data in the base.Map m1 and the base.Map m2 with the function f, for keys existing in both maps. It discards other data and keys.

For example, if

x = base.Map.fromList [("a", 5), ("b", 3)]
y = base.Map.fromList [("a", 6), ("c", 7)]

then

base.Map.intersectWith (+) x y = base.Map.singleton "a" 11

(k ->{e} a ->{e} b ->{e} c) -> base.Map k a -> base.Map k b ->{e} base.Map k c

Intersection with a combining function. The expression base.Map.intersectWithKey f m1 m2 combines data and keys in the base.Map m1 and the base.Map m2 using the function f, for keys existing in both maps. It discards other data and keys.

For example, if

x = base.Map.fromList [(1, 5), (2, 4)]
y = base.Map.fromList [(1, 6), (3, 7)]

then

base.Map.intersectWith (k v1 v2 -> k + v1 + v2) x y = base.Map.singleton 1 12

base.Map k v -> base.Map k v -> base.Map k v

Left-biased union of two maps. Contains data from both maps, and uses values from the first map if there are duplicate keys.

For example, if

x = base.Map.fromList [(5, "a"), (3, "b")]
y = base.Map.fromList [(5, "A"), (7, "C")]

then

base.Map.union x y = base.Map.fromList [(5, "a"), (3, "b"), (7, "C")]

(v ->{e} v ->{e} v) -> base.Map k v -> base.Map k v ->{e} base.Map k v

Union of two maps. Contains data from both maps, and uses the given function to combine values if there are duplicate keys.

For example, if

x = base.Map.fromList [(5, "a"), (3, "b")]
y = base.Map.fromList [(5, "A"), (7, "C")]

then

base.Map.unionWith (++) x y = base.Map.fromList [(5, "aA"), (3, "b"), (7, "C")]

(k ->{e} v ->{e} v ->{e} v) -> base.Map k v -> base.Map k v ->{e} base.Map k v

Union of two maps. Contains data from both maps, and uses the given function to combine values, and the key, for any duplicate key.

For example, if

x = base.Map.fromList [(5, "a"), (3, "b")]
y = base.Map.fromList [(5, "A"), (7, "C")]
f key left right = base.Nat.toText key ++ ":" ++ left ++ "|" ++ right

then

base.Map.unionWithKey f x y = base.Map.fromList [(5, "5:a|A"), (3, "b"), (7, "C")]

Comparing maps

base.Map k v -> base.Map k v -> base.Boolean

Checks if two maps are equal. They're equal precisely when they have all the same keys, and each key has the same value in one map as it does in the other.

Map traversal

(a ->{e} b ->{e} a) -> a -> base.Map k b ->{e} a

The expression base.Map.foldLeft f z m folds the values in the base.Map m using the left-associative binary operator f starting with z. The values in the base.Map are passed to f in key-ascending order.

For example, if

m = base.Map.fromList [(1, "a"), (2, "b"), (3, "c")]
f acc v = acc ++ v

then

base.Map.foldLeft f "x" m = "xabc"

(a ->{e} k ->{e} b ->{e} a) -> a -> base.Map k b ->{e} a

The expression base.Map.foldLeftWithKey f z m folds the keys and values in the base.Map m using the left-associative operator f starting with z. The values in the base.Map are passed to the function f in key-ascending order.

For example, if

m = base.Map.fromList [(1, "a"), (2, "b"), (3, "c")]
f acc k v = acc ++ base.List.replicate k v

then

base.Map.foldLeftWithKey f "x" m = "xabbccc"

(a ->{e} b ->{e} b) -> b -> base.Map k a ->{e} b

The expression base.Map.foldRight f z m folds the values in the base.Map m using the right-associative binary operator f starting with z. The values in the base.Map are passed to f in key-ascending order.

For example, if

m = base.Map.fromList [(1, "a"), (2, "b"), (3, "c")]
f v acc = v ++ acc

then

base.Map.foldRight f "x" m = "abcx"

(k ->{e} a ->{e} b ->{e} b) -> b -> base.Map k a ->{e} b

The expression base.Map.foldRightWithKey f z m folds the keys and values in the base.Map m using the right-associative operator f starting with z. The values in the base.Map are passed to the function f in key-ascending order.

For example, if

m = base.Map.fromList [(1, "a"), (2, "b"), (3, "c")]
f k v acc = base.List.replicate k v acc

then

base.Map.foldRightWithKey f "x" m = "abbcccx"

(a ->{e} b) -> base.Map k a ->{e} base.Map k b

The expression base.Map.map f m applies the function f to all values in the base.Map m returning a new base.Map with the results. If the input base.Map contains a value v under a key k, then the output base.Map will contain f v under the key k.

For example, if

m = base.Map.fromList [(1, "abc"), (2, "xyz")]

then

base.Map.map base.Text.reverse m = base.Map.fromList [(1, "cba"), (2, "zyx")]

(k1 ->{g} k2) -> base.Map k1 v ->{g} base.Map k2 v

The expression base.Map.mapKeys f m creates a new base.Map by applying f to each key of the base.Map m. The size of the result may be smaller if f maps two or more distinct keys to the same new key. In this case the value at the greatest (by universal equality) of the original keys is retained.

For example, if

m = base.Map.fromList [("abc", 1), ("xyz", 2)]

then

base.Map.mapKeys base.Text.reverse m = base.Map.fromList [("cba", 1), ("zyx", 2)]
base.Map.mapKeys (k -> 7) m = base.Map.singleton 7 2

(v ->{g} v ->{g} v) -> (k1 ->{g} k2) -> base.Map k1 v ->{g} base.Map k2 v

The expression base.Map.mapKeysWith c f m creates a new base.Map by applying f to each key of the base.Map m. The size of the result may be smaller if f maps two or more distinct keys to the same new key. In this case the associated values will be combined using the function c. The value under the greater (by universal equality) of the original keys is passed as the first argument to c.

For example, if

m = base.Map.fromList [("abc", 1), ("xyz", 2)]

then

base.Map.mapKeysWith (+) base.Text.reverse m = base.Map.fromList [("cba", 1), ("zyx", 2)]
base.Map.mapKeysWith (+) (k -> 7) m = base.Map.singleton 7 3

(k ->{e} a ->{e} b) -> base.Map k a ->{e} base.Map k b

The expression base.Map.mapWithKey f m applies the function f to all values and keys in the base.Map m returning a new base.Map with the results as the values. If the input base.Map contains a value v under a key k, then the output base.Map will contain the value f k v under the key k.

For example, if

m = base.Map.fromList [(1, 9), (2, 3)]

then

base.Map.map (+) m = base.Map.fromList [(1, 10), (2, 5)]

Splitting and breaking maps

base.Map k v -> base.Optional ((k, v), base.Map k v)

Finds the minimal key in the base.Map and removes it. Returns the key, the value under the key, and the remainder of the base.Map with that key removed. If the base.Map is empty, returns base.Optional and the unchanged base.Map.

For example, if

a = base.Map.fromList [(5, "a"), (3, "b")]

then

breakOffMin a = (Some (3, "b"), base.Map.singleton (5, "a"))

base.Map k v -> base.Optional ((k, v), base.Map k v)

Finds the maximal key in the base.Map and removes it. Returns the key, the value under the key, and the remainder of the base.Map with that key removed. If the base.Map is empty, returns base.Optional and the unchanged base.Map.

For example, if

a = base.Map.fromList [(5, "a"), (3, "b")]

then

breakOffMax a = (Some (5, "a"), base.Map.singleton (3, "b"))

k -> base.Map k a -> (base.Map k a, base.Map k a)

Splits the base.Map in two parts around a given key. The expression base.Map.split k m is a pair (m1, m2) where all keys in m1 are smaller than k and all keys in m2 are larger than k. Any key equal to k is discarded.

Conversions

[(k, v)] -> base.Map k v

Constructs a base.Map from a base.List of key/value pairs. If the base.List contains more than one value for the same key, the last value for the key is used. See base.Map.fromListWith and base.Map.fromListWithKey if you want to combine values under duplicate keys in some other way.

base.Map k a -> [k]

Returns all the keys in the base.Map, as a base.List.

base.Map k v -> [(k, v)]

Gets the list of key-value pairs in the base.Map.

base.Map k v -> [v]

Gets all the values in a base.Map, as a base.List.

fromList ->

doc
tests
base.Map.fromList : [(k, v)] -> base.Map k v
base.Map.fromList = cases
  []               -> base.Map.internal.Tip
  [(kx, x)]        -> base.Map.singleton kx x
  (kx0, x0) +: xs0 ->
    use base.Map singleton
    use base.Map.internal link putMax
    notOrdered kx = cases
      []           -> false
      (ky, _) +: _ ->
        use base.Universal >=
        kx >= ky
    fromList' t0 xs =
      base.List.foldLeft (cases t (k, x) -> base.Map.insert k x t) t0 xs
    create s = cases
      []             -> (base.Map.internal.Tip, [], [])
      xs@(xp +: xss) ->
        use base.Universal ==
        if s == +1 then
          (kx, x) = xp
          if notOrdered kx xss then (singleton kx x, [], xss)
          else (singleton kx x, xss, [])
        else
          use base.Int shiftRight
          match create (shiftRight s 1) xs with
            res@(_, [], _)              -> res
            (l, [(ky, y)], zs)          -> (putMax ky y l, [], zs)
            (l, ys@((ky, y) +: yss), _) ->
              if notOrdered ky yss then (l, [], ys)
              else
                (r, zs, ws) = create (shiftRight s 1) yss
                (link ky y l r, zs, ws)
    go = cases
      _ t []                  -> t
      _ t [(kx, x)]           -> putMax kx x t
      s l xs@((kx, x) +: xss) ->
        if notOrdered kx xss then fromList' l xs
        else
          match create s xss with
            (r, ys, []) -> go (base.Int.shiftLeft s 1) (link kx x l r) ys
            (r, _, ys)  -> fromList' (link kx x l r) ys
    if notOrdered kx0 xs0 then fromList' (singleton kx0 x0) xs0
    else go +1 (singleton kx0 x0) xs0

  (k ->{e} v ->{e} v ->{e} v)
  -> base.Map k v
  -> base.Map k v
  ->{e} base.Map k v
base.Map.unionWithKey f t1 t2 =
  use base.Map.internal Bin Tip
  match (t1, t2) with
    (_, Tip)               -> t1
    (_, Bin _ k x Tip Tip) -> base.Map.internal.putWithKeyR f k x t1
    (Bin _ k x Tip Tip, _) -> base.Map.putWithKey f k x t2
    (Tip, _)               -> t2
    (Bin _ k1 x1 l1 r1, _) ->
      use base.Map unionWithKey
      use base.Map.internal link
      (l2, mb, r2) = base.Map.internal.splitLookup k1 t2
      l1l2 = unionWithKey f l1 l2
      r1r2 = unionWithKey f r1 r2
      match mb with
        base.Optional.None    -> link k1 x1 l1l2 r1r2
        base.Optional.Some x2 -> link k1 (f k1 x1 x2) l1l2 r1r2
