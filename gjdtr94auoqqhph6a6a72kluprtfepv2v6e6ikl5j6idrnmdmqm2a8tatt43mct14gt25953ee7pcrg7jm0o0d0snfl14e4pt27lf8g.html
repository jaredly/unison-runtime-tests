type base.Weighted a
  = Weight base.Nat 'base.Weighted a
  | Yield a (base.Weighted a)
  | Fail

<|>

(base.Weighted.<|>) : base.Weighted a -> base.Weighted a -> base.Weighted a
(base.Weighted.<|>) = cases
  base.Weighted.Fail n                                   -> n
  (base.Weighted.Yield x m) n                            ->
    use base.Weighted <|>
    base.Weighted.Yield x (m <|> n)
  (base.Weighted.Weight w m) base.Weighted.Fail          ->
    base.Weighted.Weight w m
  (base.Weighted.Weight w m) (base.Weighted.Yield x n)   ->
    use base.Weighted <|>
    base.Weighted.Yield x (base.Weighted.Weight w m <|> n)
  (base.Weighted.Weight w m) (base.Weighted.Weight w' n) ->
    use base.Nat drop
    use base.Universal <
    use base.Weighted <|> Weight
    if w < w' then Weight w '(!m <|> Weight (drop w' w) n)
    else
      use base.Universal ==
      if w == w' then Weight w '(!m <|> !n)
      else Weight w '(Weight (drop w w') m <|> !n)

base.Weighted.append : base.Weighted a -> base.Weighted a -> base.Weighted a
base.Weighted.append w1 w2 =
  use base.Weighted Weight Yield append
  match w1 with
    Weight n k         -> Weight n '(append !k w2)
    Yield a k          -> Yield a (append k w2)
    base.Weighted.Fail -> w2

filter

base.Weighted.filter :
  (a ->{e} base.Boolean) -> base.Weighted a ->{e} base.Weighted a
base.Weighted.filter p = cases
  base.Weighted.Yield a k  ->
    (if p a then base.Weighted.Yield a else base.id) (base.Weighted.filter p k)
  base.Weighted.Weight n k ->
    base.Weighted.Weight n '(base.Weighted.filter p !k)
  w                        -> w

flatMap

base.Weighted.flatMap :
  (a -> base.Weighted b) -> base.Weighted a -> base.Weighted b
base.Weighted.flatMap f = cases
  base.Weighted.Fail       -> base.Weighted.Fail
  base.Weighted.Yield x m  ->
    use base.Weighted <|>
    f x <|> base.Weighted.flatMap f m
  base.Weighted.Weight w m ->
    base.Weighted.Weight w '(base.Weighted.flatMap f !m)

fromList

base.Weighted.fromList : [a] -> base.Weighted a
base.Weighted.fromList = cases
  []      -> base.Weighted.Fail
  a +: as ->
    use base.Weighted <|>
        base.Weighted.yield a
    <|> base.Weighted.weight 1 '(base.Weighted.fromList as)

ints

base.Weighted.ints : base.Weighted base.Int
base.Weighted.ints =
  use base.Weighted <|> yield
  go n =
    use base.Int negate
    use base.Universal >
        yield n
    <|> base.Weighted.weight
          1 '(go (if n > +0 then negate n else base.Int.increment (negate n)))
      base.List.foldLeft
        (a n -> a <|> yield n)
        base.Weighted.Fail
        [+0, +1, -1, base.Int.maxInt, base.Int.minInt]
  <|> go +2

lists

base.Weighted.lists : base.Weighted a -> base.Weighted [a]
base.Weighted.lists w =
  use base.List +:
  use base.Weighted <|>
      base.Weighted.yield []
  <|> base.Weighted.weight
        1 '(base.Weighted.mergeWith (+:) w (base.Weighted.lists w))

map

base.Weighted.map : (a ->{e} b) -> base.Weighted a ->{e} base.Weighted b
base.Weighted.map f = cases
  base.Weighted.Fail       -> base.Weighted.Fail
  base.Weighted.Yield x w  -> base.Weighted.Yield (f x) (base.Weighted.map f w)
  base.Weighted.Weight a w -> base.Weighted.weight a '(base.Weighted.map f !w)

nats

base.Weighted.nats : base.Weighted base.Nat
base.Weighted.nats =
  go n =
    use base.Nat +
    use base.Weighted <|>
    base.Weighted.yield n <|> base.Weighted.weight 1 '(go (n + 1))
  go 0

sample

base.Weighted.sample : base.Nat -> base.Weighted a -> [a]
base.Weighted.sample n wsa =
  use base.Universal >
  if n > 0 then
    use base.Weighted sample
    match wsa with
      base.Weighted.Fail       -> []
      base.Weighted.Yield a ms ->
        use base.List +:
        a +: sample (base.Nat.drop n 1) ms
      base.Weighted.Weight _ w -> sample n !w
  else []
