  use base.Nat + drop
  use base.Universal >

base.Nat.factorial : base.Nat -> base.Nat
base.Nat.factorial n =
  use base.Universal <
  go acc n =
    use base.Universal ==
    if n == 1 then acc
    else
      use base.Nat *
      go (acc * n) (base.Nat.drop n 1)
  if n < 2 then 1 else go 1 n

base.Nat.msb : base.Nat -> base.Optional base.Nat
base.Nat.msb n =
  use base.Universal ==
  go d n =
    use base.Universal >
    if n > 0 then
      use base.Nat +
      go (d + 1) (base.Nat.shiftRight n 1)
    else d
  if n == 0 then base.Optional.None
  else base.Optional.Some (base.Nat.drop (go 0 n) 1)
