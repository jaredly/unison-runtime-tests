pick

base.test.pick : ['{base.test.Gen} a] -> '{base.test.Gen} a
base.test.pick gs =
  use base.test Gen
  merge : '{Gen} a -> '{Gen} a -> '{Gen} a
  merge g1 g2 _ =
    use base |>
    use base.Weighted <|>
    use base.test.Gen toWeighted
    toWeighted g1 <|> toWeighted g2 |> base.test.Gen.sample
  base.List.foldb base.id merge base.test.gen.empty gs

base.test.runs :

base.test.runs n ts =
  use base <|
  base.test.run <| base.test.tests (base.test.sample n ts)

base.test.sample : base.Nat -> '{e, base.test.Gen} a ->{e} [a]

base.test.tests : [base.test.Test] -> base.test.Test
base.test.tests = cases
  []      -> base.test.ok
  h +: ts -> base.List.foldLeft base.test.both h ts
