Home : base . Bag
base.Bag.addN : base.Nat -> k -> base.Bag k -> base.Bag k
base.Bag.addN n k b =
  use base.Bag +
  b + base.Bag.scale n (base.Bag.singleton k)

base.Bag.addN.test =
  use base.Universal >=
  use base.test.gen.Char ascii
  base.test.runs
    100 'let
      b = !(base.Bag.tests.bagOf ascii)
      c = !ascii
      n = !base.test.gen.nat
      base.test.expect (base.Bag.count c (base.Bag.addN n c b) >= n)
