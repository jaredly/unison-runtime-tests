balance

base.Map.internal.balance :
  k -> v -> base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.balance k x l r =
  use base Map
  use base.Map size
  use base.Map.internal Bin bin ratio
  use base.Nat * +
  use base.Universal < <=
  rotateL : a -> b -> Map a b -> Map a b -> Map a b
  rotateL k x l = cases
    r@(Bin _ _ _ ly ry) | size ly < (ratio * size ry) -> singleL k x l r
    r                                                 -> doubleL k x l r
  rotateR : a -> b -> Map a b -> Map a b -> Map a b
  rotateR k x = cases
    l@(Bin _ _ _ ly ry) | size ry < (ratio * size ly) -> singleR k x l
    l                                                 -> doubleR k x l
  singleL : a -> b -> Map a b -> Map a b -> Map a b
  singleL k1 x1 t1 = cases Bin _ k2 x2 t2 t3 -> bin k2 x2 (bin k1 x1 t1 t2) t3
  singleR : a -> b -> Map a b -> Map a b -> Map a b
  singleR k1 x1 = cases
    Bin _ k2 x2 t1 t2 -> t3 -> bin k2 x2 t1 (bin k1 x1 t2 t3)
  doubleL : a -> b -> Map a b -> Map a b -> Map a b
  doubleL k1 x1 t1 = cases
    Bin _ k2 x2 (Bin _ k3 x3 t2 t3) t4 ->
      bin k3 x3 (bin k1 x1 t1 t2) (bin k2 x2 t3 t4)
  doubleR : a -> b -> Map a b -> Map a b -> Map a b
  doubleR k1 x1 = cases
    Bin _ k2 x2 t1 (Bin _ k3 x3 t2 t3) ->
      t4 -> bin k3 x3 (bin k2 x2 t1 t2) (bin k1 x1 t3 t4)
  sizeL = size l
  sizeR = size r
  sizeX = sizeL + sizeR + 1
  if sizeL + sizeR <= 1 then Bin sizeX k x l r
  else
    use base.Map.internal delta
    use base.Universal >
    if sizeR > (delta * sizeL) then rotateL k x l r
    else if sizeL > (delta * sizeR) then rotateR k x l r else Bin sizeX k x l r

balanceL

base.Map.internal.balanceL :
  k -> v -> base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.balanceL k x l = cases
  base.Map.internal.Tip                ->
    use base.Map singleton
    use base.Map.internal Bin Tip
    match l with
      Tip                                                            ->
        singleton k x
      Bin _ _ _ Tip Tip                                              ->
        Bin 2 k x l Tip
      Bin _ lk lx Tip (Bin _ lrk lrx _ _)                            ->
        Bin 3 lrk lrx (singleton lk lx) (singleton k x)
      Bin _ lk lx ll@(Bin _ _ _ _ _) Tip                             ->
        Bin 3 lk lx ll (singleton k x)
      Bin ls lk lx ll@(Bin lls _ _ _ _) lr@(Bin lrs lrk lrx lrl lrr) ->
        use base.Nat * +
        use base.Universal <
        if lrs < (base.Map.internal.ratio * lls) then
          Bin (1 + ls) lk lx ll (Bin (1 + lrs) k x lr Tip)
        else
          use base.Map size
          Bin
            (1 + ls)
            lrk
            lrx
            (Bin (1 + lls + size lrl) lk lx ll lrl)
            (Bin (1 + size lrr) k x lrr Tip)
  r@(base.Map.internal.Bin rs _ _ _ _) ->
    use base.Map.internal Bin Tip
    use base.Nat +
    match l with
      Tip                -> Bin (1 + rs) k x Tip r
      Bin ls lk lx ll lr ->
        use base.Nat *
        use base.Universal >
        if ls > (base.Map.internal.delta * rs) then
          match (ll, lr) with
            (Bin lls _ _ _ _, Bin lrs lrk lrx lrl lrr) ->
              use base.Universal <
              if lrs < (base.Map.internal.ratio * lls) then
                Bin (1 + ls + rs) lk lx ll (Bin (1 + rs + lrs) k x lr r)
              else
                use base.Map size
                Bin
                  (1 + ls + rs)
                  lrk
                  lrx
                  (Bin (1 + lls + size lrl) lk lx ll lrl)
                  (Bin (1 + rs + size lrr) k x lrr r)
            (_, _)                                     ->
              base.bug "failure in balanceL"
        else Bin (1 + ls + rs) k x l r

balanceR

base.Map.internal.balanceR :
  k -> v -> base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.balanceR k x l r =
  use base.Map size
  use base.Map.internal Bin Tip ratio
  use base.Nat * +
  use base.Universal <
  match l with
    Tip            ->
      match r with
        Tip                                                                                  ->
          Bin 1 k x Tip Tip
        Bin _ _ _ Tip Tip                                                                    ->
          Bin 2 k x Tip r
        Bin _ rk rx Tip rr@(Bin _ _ _ _ _)                                                   ->
          Bin 3 rk rx (Bin 1 k x Tip Tip) rr
        Bin _ rk rx (Bin _ rlk rlx _ _) Tip                                                  ->
          Bin 3 rlk rlx (Bin 1 k x Tip Tip) (Bin 1 rk rx Tip Tip)
        Bin rs rk rx rl@(Bin rls rlk rlx rll rlr) rr@(Bin rrs _ _ _ _) |
        rls < (ratio * rrs) ->
          Bin (1 + rs) rk rx (Bin (1 + rls) k x Tip rl) rr
        Bin rs rk rx rl@(Bin rls rlk rlx rll rlr) rr@(Bin rrs _ _ _ _)                       ->
          Bin
            (1 + rs)
            rlk
            rlx
            (Bin (1 + size rll) k x Tip rll)
            (Bin (1 + rrs + size rlr) rk rx rlr rr)
    Bin ls _ _ _ _ ->
      use base.Universal >
      match r with
        Tip                                                      ->
          Bin (1 + ls) k x l Tip
        Bin rs rk rx rl rr | rs > (base.Map.internal.delta * ls) ->
          match (rl, rr) with
            (Bin rls rlk rlx rll rlr, Bin rrs _ _ _ _) | rls < (ratio * rrs) ->
              Bin (1 + ls + rs) rk rx (Bin (1 + ls + rls) k x l rl) rr
            (Bin rls rlk rlx rll rlr, Bin rrs _ _ _ _)                       ->
              Bin
                (1 + ls + rs)
                rlk
                rlx
                (Bin (1 + ls + size rll) k x l rll)
                (Bin (1 + rrs + size rlr) rk rx rlr rr)
            (_, _)                                                           ->
              base.bug "Failure in balanceR"
        Bin rs rk rx rl rr                                       ->
          Bin (1 + ls + rs) k x l r

bin

base.Map.internal.bin : k -> v -> base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.bin k x l r =
  use base.Map size
  use base.Nat +
  base.Map.internal.Bin (size l + size r + 1) k x l r

delta

base.Map.internal.delta : base.Nat
base.Map.internal.delta = 3

glue

base.Map.internal.glue : base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.glue = cases
  base.Map.internal.Tip                    -> base.id
  l@(base.Map.internal.Bin sl kl xl ll lr) ->
    use base.Map.internal Bin
    use base.Universal >
    cases
      base.Map.internal.Tip            -> l
      r@(Bin sr kr xr rl rr) | sl > sr ->
        base.Map.internal.MaxView.MaxView km m l' =
          base.Map.internal.maxViewSure kl xl ll lr
        base.Map.internal.balanceR km m l' r
      r@(Bin sr kr xr rl rr)           ->
        base.Map.internal.MinView.MinView km m r' =
          base.Map.internal.minViewSure kr xr rl rr
        base.Map.internal.balanceL km m l r'

link

base.Map.internal.link : k -> v -> base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.link kx x = cases
  base.Map.internal.Tip r                                                               ->
    base.Map.internal.putMin kx x r
  l base.Map.internal.Tip                                                               ->
    base.Map.internal.putMax kx x l
  l@(base.Map.internal.Bin sizeL ky y ly ry)
  r@(base.Map.internal.Bin sizeR kz z lz rz) ->
    use base.Map.internal delta link
    use base.Nat *
    use base.Universal <
    if delta * sizeL < sizeR then
      base.Map.internal.balanceL kz z (link kx x l lz) rz
    else
      if delta * sizeR < sizeL then
        base.Map.internal.balanceR ky y ly (link kx x ry r)
      else base.Map.internal.bin kx x l r

link2

base.Map.internal.link2 : base.Map k v -> base.Map k v -> base.Map k v
base.Map.internal.link2 l r =
  use base.Map.internal Bin Tip
  match (l, r) with
    (Tip, r)                                     -> r
    (l, Tip)                                     -> l
    (Bin sizeL kx x lx rx, Bin sizeR ky y ly ry) ->
      use base.Map.internal delta link2
      use base.Nat *
      use base.Universal <
      if delta * sizeL < sizeR then
        base.Map.internal.balanceL ky y (link2 l ly) ry
      else
        if delta * sizeR < sizeL then
          base.Map.internal.balanceR kx x lx (link2 rx r)
        else base.Map.internal.glue l r

maxViewSure

base.Map.internal.maxViewSure :
  k -> v -> base.Map k v -> base.Map k v -> base.Map.internal.MaxView k v
base.Map.internal.maxViewSure k x = cases
  l base.Map.internal.Tip                 ->
    base.Map.internal.MaxView.MaxView k x l
  l (base.Map.internal.Bin _ kr xr rl rr) ->
    use base.Map.internal.MaxView MaxView
    MaxView km xm r' = base.Map.internal.maxViewSure kr xr rl rr
    MaxView km xm (base.Map.internal.balanceL k x l r')

minViewSure

base.Map.internal.minViewSure :
  k -> v -> base.Map k v -> base.Map k v -> base.Map.internal.MinView k v
base.Map.internal.minViewSure k x = cases
  base.Map.internal.Tip               -> base.Map.internal.MinView.MinView k x
  base.Map.internal.Bin _ kl xl ll lr ->
    use base.Map.internal.MinView MinView
    r ->
      (match base.Map.internal.minViewSure kl xl ll lr with
        MinView km xm l' -> MinView km xm (base.Map.internal.balanceR k x l' r))

putMax

base.Map.internal.putMax : k -> v -> base.Map k v -> base.Map k v
base.Map.internal.putMax kx x = cases
  base.Map.internal.Tip            -> base.Map.singleton kx x
  base.Map.internal.Bin _ ky y l r ->
    base.Map.internal.balanceR ky y l (base.Map.internal.putMax kx x r)

putMin

base.Map.internal.putMin : k -> v -> base.Map k v -> base.Map k v
base.Map.internal.putMin kx x = cases
  base.Map.internal.Tip            -> base.Map.singleton kx x
  base.Map.internal.Bin _ ky y l r ->
    base.Map.internal.balanceL ky y (base.Map.internal.putMin kx x l) r

putWithKeyR

base.Map.internal.putWithKeyR :
  (k ->{e} v ->{e} v ->{e} v) -> k -> v -> base.Map k v ->{e} base.Map k v
base.Map.internal.putWithKeyR f kx x = cases
  base.Map.internal.Tip             -> base.Map.singleton kx x
  base.Map.internal.Bin sy ky y l r ->
    use base.Map.internal putWithKeyR
    match base.Universal.ordering kx ky with
      base.Ordering.Less    ->
        base.Map.internal.balanceL ky y (putWithKeyR f kx x l) r
      base.Ordering.Greater ->
        base.Map.internal.balanceR ky y l (putWithKeyR f kx x r)
      base.Ordering.Equal   -> base.Map.internal.Bin sy ky (f ky y x) l r

putWithR

base.Map.internal.putWithR :

base.Map.internal.putWithR f kx x = cases
  base.Map.internal.Tip             -> base.Map.singleton kx x
  base.Map.internal.Bin sy ky y l r ->

ratio

splitLookup

base.Map.internal.splitLookup :
  k -> base.Map k v -> (base.Map k v, base.Optional v, base.Map k v)
base.Map.internal.splitLookup k = cases
  base.Map.internal.Tip            ->
    use base.Map.internal Tip
    (Tip, base.Optional.None, Tip)
  base.Map.internal.Bin _ kx x l r ->
    use base.Map.internal link splitLookup
    match base.Universal.ordering k kx with
      base.Ordering.Less    ->
        (lt, z, gt) = splitLookup k l
        (lt, z, link kx x gt r)
      base.Ordering.Greater ->
        (lt, z, gt) = splitLookup k r
        (link kx x l lt, z, gt)
      base.Ordering.Equal   -> (l, base.Optional.Some x, r)
