type base.Heap k v = Heap base.Nat k v [base.Heap k v]

children

base.Heap.children : base.Heap k v -> [base.Heap k v]
base.Heap.children = cases base.Heap.Heap _ _ _ cs -> cs

fromKeys

base.Heap.fromKeys : [a] -> base.Optional (base.Heap a a)
base.Heap.fromKeys as = base.Heap.fromList (base.List.map (a -> (a, a)) as)

fromList

base.Heap.fromList : [(k, v)] -> base.Optional (base.Heap k v)
base.Heap.fromList kvs =
  use base.Optional None Some
  op a b =
    match a with
      None   -> b
      Some a ->
        match b with
          None   -> Some a
          Some b -> Some (base.Heap.union a b)
  single = cases (k, v) -> Some (base.Heap.singleton k v)
  base.List.foldb single op None kvs

max

base.Heap.max : base.Heap k v -> (k, v)
base.Heap.max = cases base.Heap.Heap _ k v _ -> (k, v)

maxKey

base.Heap.maxKey : base.Heap k v -> k
base.Heap.maxKey = cases base.Heap.Heap _ k _ _ -> k

pop

base.Heap.pop : base.Heap k v -> base.Optional (base.Heap k v)
base.Heap.pop h =
  use base.Optional None Some
  go h subs =
    use base.List size
    use base.Universal ==
    if size subs == 0 then h
    else
      use base.Heap union
      use base.List unsafeAt
      if size subs == 1 then union h (unsafeAt 0 subs)
      else
        union
          (union h (unsafeAt 0 subs))
          (go (unsafeAt 1 subs) (base.List.drop 2 subs))
  match base.List.uncons (base.Heap.children h) with
    None            -> None
    Some (s0, subs) -> Some (go s0 subs)

singleton

base.Heap.singleton : k -> v -> base.Heap k v
base.Heap.singleton k v = base.Heap.Heap 1 k v []

size

base.Heap.size : base.Heap k v -> base.Nat
base.Heap.size = cases base.Heap.Heap n _ _ _ -> n

sort

base.Heap.sort : [a] -> [a]
base.Heap.sort as =
  use base |>
  base.Heap.sortDescending as |> base.List.reverse

sortDescending

base.Heap.sortDescending : [a] -> [a]
base.Heap.sortDescending as =
  use base |>
  step = cases
    base.Optional.None   -> base.Optional.None
    base.Optional.Some h ->
      base.Optional.Some (base.Heap.max h, base.Heap.pop h)
  base.List.unfold (base.Heap.fromKeys as) step |> base.List.map base.Tuple.at1

union

base.Heap.union : base.Heap k v -> base.Heap k v -> base.Heap k v
base.Heap.union h1 h2 =
  use base.Heap Heap
  use base.List +:
  use base.Nat +
  use base.Universal >=
  (Heap n k1 v1 hs1, Heap m k2 v2 hs2) = (h1, h2)
  if k1 >= k2 then Heap (n + m) k1 v1 (h2 +: hs1)
  else Heap (n + m) k2 v2 (h1 +: hs2)
