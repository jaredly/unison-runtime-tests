forAll'

base.test.internals.v1.Test.forAll' :
  base.Nat
  -> base.test.internals.v1.Domain a
  -> (a ->{e} base.Boolean)
  ->{e} base.test.Test
base.test.internals.v1.Test.forAll' maxSize domain property =
  use base <|
  check xs s =
    use base.test.internals.v1.Test finished
    base.List.map
      (cases
        (c, i) ->
          if property c then
            finished (base.test.internals.v1.Test.Status.Expected s)
          else
            use base.Text ++
            base.test.label
              ("test case " ++ base.Nat.toText i)
              (finished base.test.internals.v1.Test.Status.Failed))
      (base.List.indexed xs)
     base.List.foldb base.id base.test.both base.test.prove
  <| (match domain with
       base.test.internals.v1.Domain.Small xs ->
         check
           (base.List.take maxSize xs)
           base.test.internals.v1.Test.Success.Proved
       base.test.internals.v1.Domain.Large _  ->
         check
           (base.test.internals.v1.Domain.sample maxSize domain)
           (base.test.internals.v1.Test.Success.Passed 1))
