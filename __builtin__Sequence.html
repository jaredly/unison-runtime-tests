  []        -> []
  x' +: xs' ->
    use base.List +:
    use base.List.Nonempty +|
    go p z = cases
      []      -> ([], [])
      x +: xs ->
        (ys, zs) = go p x xs
        if p z x then (x +: ys, zs) else ([], x +| ys +: zs)
    let
      (ys', zs') = go p' x' xs'
      x' +| ys' +: zs'
